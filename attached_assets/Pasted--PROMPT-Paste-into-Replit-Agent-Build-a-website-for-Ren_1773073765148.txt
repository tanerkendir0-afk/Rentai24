### 1. Connect Your AI Model
Replace the mock responses in `/api/chat/route.ts`:

```typescript
// /api/chat/route.ts
export async function POST(req: Request) {
  const { message, agentType, conversationHistory } = await req.json();

  // System prompts per agent type
  const systemPrompts: Record<string, string> = {
    customer_support: "You are a professional customer support AI agent for RentAI 24 demo. You handle complaints, track orders, process refunds, and answer FAQs. Be helpful, empathetic, and efficient.",
    sales: "You are a sales development AI agent for RentAI 24 demo. You generate leads, write outreach emails, and schedule meetings. Be persuasive but professional.",
    social_media: "You are a social media manager AI agent for RentAI 24 demo. You plan content, write posts, and analyze engagement. Be creative and trend-aware.",
    bookkeeping: "You are a bookkeeping AI agent for RentAI 24 demo. You handle invoices, expenses, and financial reports. Be precise and detail-oriented.",
    scheduling: "You are a scheduling AI agent for RentAI 24 demo. You manage appointments, send reminders, and handle rescheduling. Be organized and proactive.",
    hr: "You are an HR & recruiting AI agent for RentAI 24 demo. You screen resumes, schedule interviews, and manage onboarding. Be thorough and fair.",
  };

  const response = await fetch(process.env.AI_API_URL!, {
    method: 'POST',
    headers: {
      'Authorization': `Bearer ${process.env.AI_API_KEY}`,
      'Content-Type': 'application/json',
    },
    body: JSON.stringify({
      model: process.env.AI_MODEL_ID,
      system: systemPrompts[agentType] || "You are a helpful AI agent demo for RentAI 24.",
      messages: [
        ...conversationHistory,
        { role: 'user', content: message }
      ],
      max_tokens: 500,
    }),
  });

  const data = await response.json();
  return Response.json({ reply: data.content[0].text });
}
```
