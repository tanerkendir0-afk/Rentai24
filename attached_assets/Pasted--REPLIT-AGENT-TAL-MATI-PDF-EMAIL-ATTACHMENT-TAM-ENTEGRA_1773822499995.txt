Projemde `rentai24-pdf-tools/` klasörü var. İçinde hazır PDF oluşturma, email gönderme
ve tool registry kodları var. Bunları mevcut agent engine'e entegre etmeni istiyorum.

ÖNEMLİ: Projemde hem Claude (Anthropic) hem OpenAI (GPT) modelleri kullanılıyor.
Tool calling her iki API için de çalışmalı. İkisinin tool call formatı farklı,
bu yüzden model-agnostic bir abstraction layer yaz.

Önce `rentai24-pdf-tools/docs/ENTEGRASYON_REHBERI.md` dosyasını oku, tüm detaylar orada.

```python
def format_tools_for_provider(tools: list, provider: str) -> list:
    """
    Ortak tool tanımlarını provider-specific formata dönüştürür.

    provider: "anthropic" veya "openai"
    """
    if provider == "anthropic":
        # Claude formatı: input_schema
        return [
            {
                "name": t["name"],
                "description": t["description"],
                "input_schema": t["parameters"]
            }
            for t in tools
        ]

    elif provider == "openai":
        # OpenAI formatı: type=function, function={name, description, parameters}
        return [
            {
                "type": "function",
                "function": {
                    "name": t["name"],
                    "description": t["description"],
                    "parameters": t["parameters"]
                }
            }
            for t in tools
        ]

    else:
        raise ValueError(f"Desteklenmeyen provider: {provider}")
```

### 5. MODEL-AGNOSTİK TOOL CALL LOOP

Bu en kritik kısım. Claude ve OpenAI'nin tool call/response formatı farklı.
Tek bir `process_agent_message` fonksiyonu her ikisini de handle etmeli:

```python
# ============================================================
# services/llm_service.py — Model-agnostic LLM servisi
# ============================================================

import json
import os
from anthropic import Anthropic
from openai import OpenAI

anthropic_client = Anthropic(api_key=os.environ.get("ANTHROPIC_API_KEY"))
openai_client = OpenAI(api_key=os.environ.get("OPENAI_API_KEY"))


def execute_tool(tool_name: str, tool_input: dict, branding: dict = None) -> dict:
    """Tool'u çalıştır — provider'dan bağımsız."""
    from rentai24_pdf_tools.tools.pdf_generator import handle_generate_pdf
    from rentai24_pdf_tools.services.email_service import handle_send_email

    if tool_name == "generate_pdf":
        return handle_generate_pdf(tool_input, branding=branding)
    elif tool_name == "send_email":
        return handle_send_email(tool_input)
    else:
        return {"success": False, "error": f"Bilinmeyen tool: {tool_name}"}


# ============================================================
# CLAUDE (Anthropic) — Tool Call Loop
# ============================================================
async def process_with_claude(system_prompt: str, user_message: str,
                               tools: list, branding: dict = None,
                               model: str = "claude-sonnet-4-20250514") -> str:
    """Claude API ile tool call loop."""

    formatted_tools = format_tools_for_provider(tools, "anthropic")
    messages = [{"role": "user", "content": user_message}]

    response = anthropic_client.messages.create(
        model=model,
        max_tokens=4096,
        system=system_prompt,
        tools=formatted_tools,
        messages=messages
    )

    # Tool call loop
    while response.stop_reason == "tool_use":
        assistant_content = response.content
        tool_results = []

        for block in assistant_content:
            if block.type == "tool_use":
                result = execute_tool(block.name, block.input, branding=branding)
                tool_results.append({
                    "type": "tool_result",
                    "tool_use_id": block.id,
                    "content": json.dumps(result, ensure_ascii=False)
                })

        messages.append({"role": "assistant", "content": assistant_content})
        messages.append({"role": "user", "content": tool_results})

        response = anthropic_client.messages.create(
            model=model,
            max_tokens=4096,
            system=system_prompt,
            tools=formatted_tools,
            messages=messages
        )

    # Final text
    return "".join(block.text for block in response.content if hasattr(block, 'text'))


# ============================================================
# OPENAI (GPT) — Tool Call Loop
# ============================================================
async def process_with_openai(system_prompt: str, user_message: str,
                               tools: list, branding: dict = None,
                               model: str = "gpt-4o") -> str:
    """OpenAI API ile tool call loop."""

    formatted_tools = format_tools_for_provider(tools, "openai")
    messages = [
        {"role": "system", "content": system_prompt},
        {"role": "user", "content": user_message}
    ]

    response = openai_client.chat.completions.create(
        model=model,
        messages=messages,
        tools=formatted_tools,
        tool_choice="auto"
    )

    message = response.choices[0].message

    # Tool call loop
    while message.tool_calls:
        # Assistant mesajını ekle
        messages.append(message)

        # Her tool call'u işle
        for tool_call in message.tool_calls:
            func_name = tool_call.function.name
            func_args = json.loads(tool_call.function.arguments)

            result = execute_tool(func_name, func_args, branding=branding)

            # OpenAI formatında tool result ekle
            messages.append({
                "role": "tool",
                "tool_call_id": tool_call.id,
                "content": json.dumps(result, ensure_ascii=False)
            })

        # Tekrar çağır
        response = openai_client.chat.completions.create(
            model=model,
            messages=messages,
            tools=formatted_tools,
            tool_choice="auto"
        )
        message = response.choices[0].message

    return message.content or ""


# ============================================================
# ANA FONKSİYON — Provider'a göre yönlendir
# ============================================================
async def process_agent_message(
    agent_name: str,
    user_message: str,
    customer_id: str,
    provider: str = None,     # "anthropic" veya "openai"
    model: str = None          # model override
) -> dict:
    """
    Herhangi bir agent'ın mesajını işler.
    Provider ve model agent config'den veya parametreden gelir.

    Returns:
        {
            "text": "Agent'ın cevabı",
            "pdf": {"filename": "...", "base64": "..."} veya None,
            "email_sent": True/False
        }
    """
    # Agent config'den provider ve model belirle
    agent_config = get_agent_config(agent_name)
    provider = provider or agent_config.get("provider", "anthropic")
    model = model or agent_config.get("model", "claude-sonnet-4-20250514")

    # Müşteri branding
    branding = await get_customer_branding(customer_id)

    # System prompt
    system_prompt = get_system_prompt(agent_name)

    # Tool'ları belirle
    tools = get_tools_for_agent(agent_name)  # ALL_TOOLS veya subset

    # Provider'a göre çağır
    if provider == "anthropic":
        response_text = await process_with_claude(
            system_prompt, user_message, tools, branding, model
        )
    elif provider == "openai":
        response_text = await process_with_openai(
            system_prompt, user_message, tools, branding, model
        )
    else:
        raise ValueError(f"Desteklenmeyen provider: {provider}")

    return {"text": response_text}
```

## FORMAT FARKI ÖZETİ

İki API arasındaki temel farklar (senin bilmen için — kodu yukarıda yazdım):

```
                    CLAUDE (Anthropic)              OPENAI (GPT)
─────────────────────────────────────────────────────────────────
System prompt       system=string                   messages[0].role="system"
Tool tanımı         {name, description,             {type:"function", function:
                     input_schema:{...}}              {name, description, parameters}}
Tool call yanıtı    stop_reason="tool_use"          message.tool_calls != None
                    block.type=="tool_use"           tool_call.function.name
                    block.name, block.input          tool_call.function.arguments (JSON string!)
Tool result         role="user", type=              role="tool",
                    "tool_result",                   tool_call_id=...,
                    tool_use_id=...                  content=...
Son cevap           stop_reason="end_turn"          message.tool_calls == None
                    block.type=="text"               message.content
```

## BEKLENEN SONUÇ

1. Müşteri herhangi bir agent'a (Finn, Rex, vs.) "fatura oluştur ve mail at" dediğinde:
   - Agent (Claude VEYA GPT) → generate_pdf tool → müşteri markasıyla PDF
   - Agent → send_email tool → PDF attachment + HTML email
   - Agent → kullanıcıya cevap + PDF indirme linki

2. PDF'te Türkçe karakterler doğru
3. Email HTML formatında, attachment olarak PDF ekli
4. Her agent farklı provider/model kullanabilir (config'den ayarlanır)
5. White-label: müşteri markası, RentAI24 görünmez
