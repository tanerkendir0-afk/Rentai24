"""
Rex Sales Agent için CRM veritabanı tablolarını oluştur. Aşağıdaki SQL'i çalıştır.

ÖNEMLİ KURALLAR:
- Mevcut tabloları silme, sadece yeni tablolar ekle

SQL içeriği:

-- Enable UUID
CREATE EXTENSION IF NOT EXISTS "uuid-ossp";

-- ENUM Types
CREATE TYPE lead_source AS ENUM ('website', 'referral', 'cold', 'event', 'ad', 'social', 'partner');
CREATE TYPE customer_segment AS ENUM ('enterprise', 'mid', 'smb');
CREATE TYPE deal_stage AS ENUM ('new_lead', 'contacted', 'qualified', 'proposal_sent', 'negotiation', 'closed_won', 'closed_lost');
CREATE TYPE activity_type AS ENUM ('email_sent', 'email_received', 'call', 'meeting', 'note', 'stage_change', 'task', 'sequence_event');
CREATE TYPE sequence_status AS ENUM ('active', 'paused', 'completed', 'cancelled');

-- Contacts
CREATE TABLE rex_contacts (
    id UUID PRIMARY KEY DEFAULT uuid_generate_v4(),
    tenant_id UUID NOT NULL,
    company_name VARCHAR(255) NOT NULL,
    company_size VARCHAR(50),
    industry VARCHAR(100),
    website VARCHAR(255),
    contact_name VARCHAR(255) NOT NULL,
    email VARCHAR(255),
    phone VARCHAR(50),
    position VARCHAR(100),
    is_decision_maker BOOLEAN DEFAULT false,
    source lead_source DEFAULT 'cold',
    segment customer_segment DEFAULT 'smb',
    tags TEXT[] DEFAULT '{}',
    lead_score INT DEFAULT 0 CHECK (lead_score >= 0 AND lead_score <= 100),
    score_factors JSONB DEFAULT '{}',
    notes TEXT,
    last_contacted_at TIMESTAMPTZ,
    next_follow_up_at TIMESTAMPTZ,
    created_at TIMESTAMPTZ DEFAULT NOW(),
    updated_at TIMESTAMPTZ DEFAULT NOW(),
    CONSTRAINT unique_tenant_email UNIQUE (tenant_id, email)
);

-- Deals
CREATE TABLE rex_deals (
    id UUID PRIMARY KEY DEFAULT uuid_generate_v4(),
    tenant_id UUID NOT NULL,
    contact_id UUID NOT NULL REFERENCES rex_contacts(id) ON DELETE CASCADE,
    title VARCHAR(255) NOT NULL,
    description TEXT,
    value DECIMAL(12,2) NOT NULL DEFAULT 0,
    currency VARCHAR(3) DEFAULT 'TRY',
    monthly_recurring DECIMAL(12,2),
    stage deal_stage DEFAULT 'new_lead',
    probability INT DEFAULT 10 CHECK (probability >= 0 AND probability <= 100),
    stage_entered_at TIMESTAMPTZ DEFAULT NOW(),
    expected_close DATE,
    actual_close DATE,
    loss_reason VARCHAR(255),
    competitor_lost_to VARCHAR(255),
    assigned_to VARCHAR(100),
    products JSONB DEFAULT '[]',
    custom_fields JSONB DEFAULT '{}',
    created_at TIMESTAMPTZ DEFAULT NOW(),
    updated_at TIMESTAMPTZ DEFAULT NOW()
);

-- Activities
CREATE TABLE rex_activities (
    id UUID PRIMARY KEY DEFAULT uuid_generate_v4(),
    tenant_id UUID NOT NULL,
    contact_id UUID NOT NULL REFERENCES rex_contacts(id) ON DELETE CASCADE,
    deal_id UUID REFERENCES rex_deals(id) ON DELETE SET NULL,
    type activity_type NOT NULL,
    subject VARCHAR(255),
    body TEXT,
    scheduled_at TIMESTAMPTZ,
    completed_at TIMESTAMPTZ,
    duration_minutes INT,
    email_message_id VARCHAR(255),
    email_opened BOOLEAN DEFAULT false,
    email_clicked BOOLEAN DEFAULT false,
    email_replied BOOLEAN DEFAULT false,

    deal_id UUID REFERENCES rex_deals(id) ON DELETE SET NULL,

    status sequence_status DEFAULT 'active',
    current_step INT DEFAULT 0,
    total_steps INT NOT NULL,
    next_action_at TIMESTAMPTZ,
    paused_at TIMESTAMPTZ,
    completed_at TIMESTAMPTZ,
    emails_sent INT DEFAULT 0,
    emails_opened INT DEFAULT 0,
    emails_replied INT DEFAULT 0,
    sequence_config JSONB NOT NULL,
    created_at TIMESTAMPTZ DEFAULT NOW(),
    updated_at TIMESTAMPTZ DEFAULT NOW()

-- Stage History
CREATE TABLE rex_stage_history (
    id UUID PRIMARY KEY DEFAULT uuid_generate_v4(),
    deal_id UUID NOT NULL REFERENCES rex_deals(id) ON DELETE CASCADE,
    tenant_id UUID NOT NULL,
    from_stage deal_stage,
    to_stage deal_stage NOT NULL,
    changed_by VARCHAR(100) DEFAULT 'rex',
    notes TEXT,
    created_at TIMESTAMPTZ DEFAULT NOW()
);

-- Score History
CREATE TABLE rex_score_history (
    id UUID PRIMARY KEY DEFAULT uuid_generate_v4(),
    contact_id UUID NOT NULL REFERENCES rex_contacts(id) ON DELETE CASCADE,
    tenant_id UUID NOT NULL,
    old_score INT,
    new_score INT NOT NULL,
    score_factors JSONB NOT NULL,
    trigger_event VARCHAR(100),
    created_at TIMESTAMPTZ DEFAULT NOW()
);

-- Stage Config (shared, platform-wide)
CREATE TABLE rex_stage_config (
    stage deal_stage PRIMARY KEY,
    sla_days INT NOT NULL,
    default_probability INT NOT NULL,
    auto_actions JSONB DEFAULT '[]'

INSERT INTO rex_stage_config (stage, sla_days, default_probability, auto_actions) VALUES
    ('new_lead', 1, 10, '[{"action": "calculate_lead_score"}, {"action": "plan_first_outreach"}]'),
    ('contacted', 3, 20, '[{"action": "schedule_follow_up", "delay_hours": 48}]'),
    ('qualified', 5, 40, '[{"action": "match_products_rag"}, {"action": "prepare_proposal_brief"}]'),
    ('proposal_sent', 7, 60, '[{"action": "schedule_follow_up", "delay_hours": 72}]'),
    ('negotiation', 10, 80, '[{"action": "alert_decision_maker_contact"}]'),
    ('closed_won', 0, 100, '[{"action": "trigger_onboarding"}, {"action": "notify_finn_invoice"}]'),
    ('closed_lost', 0, 0, '[{"action": "log_loss_reason"}, {"action": "add_to_nurture"}]');

-- INDEXES
CREATE INDEX idx_contacts_tenant ON rex_contacts(tenant_id);
CREATE INDEX idx_contacts_segment ON rex_contacts(tenant_id, segment);
CREATE INDEX idx_contacts_score ON rex_contacts(tenant_id, lead_score DESC);
CREATE INDEX idx_contacts_tags ON rex_contacts USING GIN(tags);
CREATE INDEX idx_deals_tenant ON rex_deals(tenant_id);
CREATE INDEX idx_deals_contact ON rex_deals(contact_id);
CREATE INDEX idx_deals_stage ON rex_deals(tenant_id, stage);
CREATE INDEX idx_deals_pipeline ON rex_deals(tenant_id, stage, value) WHERE stage NOT IN ('closed_won', 'closed_lost');
CREATE INDEX idx_activities_tenant ON rex_activities(tenant_id);
CREATE INDEX idx_activities_contact ON rex_activities(contact_id);
CREATE INDEX idx_activities_deal ON rex_activities(deal_id) WHERE deal_id IS NOT NULL;
CREATE INDEX idx_sequences_tenant ON rex_sequences(tenant_id);
CREATE INDEX idx_sequences_active ON rex_sequences(tenant_id, status, next_action_at) WHERE status = 'active';
CREATE INDEX idx_stage_history_deal ON rex_stage_history(deal_id);
CREATE INDEX idx_score_history_contact ON rex_score_history(contact_id);

-- TRIGGERS: auto-update updated_at
CREATE OR REPLACE FUNCTION update_updated_at()
RETURNS TRIGGER AS $$
BEGIN
    NEW.updated_at = NOW();
    RETURN NEW;
END;
$$ LANGUAGE plpgsql;

CREATE TRIGGER trg_contacts_updated_at BEFORE UPDATE ON rex_contacts FOR EACH ROW EXECUTE FUNCTION update_updated_at();
CREATE TRIGGER trg_deals_updated_at BEFORE UPDATE ON rex_deals FOR EACH ROW EXECUTE FUNCTION update_updated_at();
CREATE TRIGGER trg_sequences_updated_at BEFORE UPDATE ON rex_sequences FOR EACH ROW EXECUTE FUNCTION update_updated_at();

-- TRIGGER: auto-log stage changes + update probability
CREATE OR REPLACE FUNCTION log_stage_change()
RETURNS TRIGGER AS $$
BEGIN
    IF OLD.stage IS DISTINCT FROM NEW.stage THEN
        INSERT INTO rex_stage_history (deal_id, tenant_id, from_stage, to_stage)
        VALUES (NEW.id, NEW.tenant_id, OLD.stage, NEW.stage);
        NEW.stage_entered_at = NOW();
        NEW.probability = CASE NEW.stage
            WHEN 'new_lead' THEN 10 WHEN 'contacted' THEN 20
            WHEN 'qualified' THEN 40 WHEN 'proposal_sent' THEN 60
            WHEN 'negotiation' THEN 80 WHEN 'closed_won' THEN 100
            WHEN 'closed_lost' THEN 0
        END;
        IF NEW.stage IN ('closed_won', 'closed_lost') THEN
            NEW.actual_close = CURRENT_DATE;
        END IF;
    END IF;
    RETURN NEW;
END;
$$ LANGUAGE plpgsql;

CREATE TRIGGER trg_deal_stage_change BEFORE UPDATE ON rex_deals FOR EACH ROW EXECUTE FUNCTION log_stage_change();

-- VIEWS
CREATE OR REPLACE VIEW rex_pipeline_summary AS
SELECT tenant_id, stage, COUNT(*) as deal_count, SUM(value) as total_value,
    SUM(value * probability / 100) as weighted_value, AVG(value) as avg_deal_size
FROM rex_deals WHERE stage NOT IN ('closed_won', 'closed_lost')
GROUP BY tenant_id, stage;

CREATE OR REPLACE VIEW rex_conversion_funnel AS
SELECT tenant_id,
    COUNT(*) FILTER (WHERE stage = 'closed_won') as won,
    COUNT(*) FILTER (WHERE stage = 'closed_lost') as lost,
    ROUND(COUNT(*) FILTER (WHERE stage = 'closed_won')::DECIMAL / 
        NULLIF(COUNT(*) FILTER (WHERE stage IN ('closed_won', 'closed_lost')), 0) * 100, 1) as win_rate
FROM rex_deals GROUP BY tenant_id;

Tabloları oluşturduktan sonra bana doğrulama yap: kaç tablo oluştu, ENUM'lar doğru mu, trigger'lar aktif mi.

"""
Rex CRM tablolarına Row Level Security (RLS) ekle. Bu, veritabanı seviyesinde tenant izolasyonu sağlayacak — yani bir müşterinin verileri başka müşteriye ASLA görünmeyecek.

ÖNEMLİ: Bu SQL'i superuser/admin kullanıcı ile çalıştır (normal app kullanıcısı RLS enable edemez).

SQL içeriği:

-- Tenant ID helper fonksiyonu
CREATE OR REPLACE FUNCTION current_tenant_id() 
RETURNS UUID AS $$
BEGIN
    RETURN NULLIF(current_setting('app.current_tenant_id', true), '')::UUID;
EXCEPTION
    WHEN OTHERS THEN RETURN NULL;
END;
$$ LANGUAGE plpgsql STABLE SECURITY DEFINER;

-- Admin bypass fonksiyonu
CREATE OR REPLACE FUNCTION is_admin_session()
RETURNS BOOLEAN AS $$
BEGIN
    RETURN COALESCE(current_setting('app.is_admin', true), 'false')::BOOLEAN;
EXCEPTION
    WHEN OTHERS THEN RETURN false;
END;
$$ LANGUAGE plpgsql STABLE SECURITY DEFINER;

-- RLS'i her tabloya uygula
ALTER TABLE rex_contacts ENABLE ROW LEVEL SECURITY;
ALTER TABLE rex_contacts FORCE ROW LEVEL SECURITY;
CREATE POLICY tenant_isolation_contacts ON rex_contacts
    USING (tenant_id = current_tenant_id() OR is_admin_session())
    WITH CHECK (tenant_id = current_tenant_id());

ALTER TABLE rex_deals ENABLE ROW LEVEL SECURITY;
ALTER TABLE rex_deals FORCE ROW LEVEL SECURITY;
CREATE POLICY tenant_isolation_deals ON rex_deals
    USING (tenant_id = current_tenant_id() OR is_admin_session())
    WITH CHECK (tenant_id = current_tenant_id());

ALTER TABLE rex_activities ENABLE ROW LEVEL SECURITY;
ALTER TABLE rex_activities FORCE ROW LEVEL SECURITY;
CREATE POLICY tenant_isolation_activities ON rex_activities
    USING (tenant_id = current_tenant_id() OR is_admin_session())
    WITH CHECK (tenant_id = current_tenant_id());

ALTER TABLE rex_sequences ENABLE ROW LEVEL SECURITY;
ALTER TABLE rex_sequences FORCE ROW LEVEL SECURITY;
CREATE POLICY tenant_isolation_sequences ON rex_sequences
    USING (tenant_id = current_tenant_id() OR is_admin_session())
    WITH CHECK (tenant_id = current_tenant_id());

ALTER TABLE rex_stage_history ENABLE ROW LEVEL SECURITY;
ALTER TABLE rex_stage_history FORCE ROW LEVEL SECURITY;
CREATE POLICY tenant_isolation_stage_history ON rex_stage_history
    USING (tenant_id = current_tenant_id() OR is_admin_session())
    WITH CHECK (tenant_id = current_tenant_id());

ALTER TABLE rex_score_history ENABLE ROW LEVEL SECURITY;
ALTER TABLE rex_score_history FORCE ROW LEVEL SECURITY;
CREATE POLICY tenant_isolation_score_history ON rex_score_history
    USING (tenant_id = current_tenant_id() OR is_admin_session())
    WITH CHECK (tenant_id = current_tenant_id());

-- Stage config: herkes okuyabilir, sadece admin yazabilir
ALTER TABLE rex_stage_config ENABLE ROW LEVEL SECURITY;
ALTER TABLE rex_stage_config FORCE ROW LEVEL SECURITY;
CREATE POLICY shared_read_stage_config ON rex_stage_config FOR SELECT USING (true);
CREATE POLICY admin_write_stage_config ON rex_stage_config FOR ALL USING (is_admin_session()) WITH CHECK (is_admin_session());

ÖNEMLİ NOT: Eğer uygulama PostgreSQL'e superuser (postgres) ile bağlanıyorsa, RLS bypasslanır. Bu durumda ayrı bir app kullanıcısı oluşturmalısın:

CREATE ROLE rentai24_app LOGIN PASSWORD 'güvenli_şifre_koy';
GRANT USAGE ON SCHEMA public TO rentai24_app;
GRANT SELECT, INSERT, UPDATE, DELETE ON ALL TABLES IN SCHEMA public TO rentai24_app;
GRANT USAGE, SELECT ON ALL SEQUENCES IN SCHEMA public TO rentai24_app;

Sonra .env'deki DATABASE_URL'i bu yeni kullanıcıyla güncelle.

Doğrulama: Tüm RLS policy'leri aktif mi kontrol et:
SELECT tablename, policyname FROM pg_policies WHERE schemaname = 'public';
"""

"""
Mevcut API'ye tenant izolasyon middleware'i ekle. Bu middleware her API request'inde:
